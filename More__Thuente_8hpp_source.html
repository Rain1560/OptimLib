28 
29  public:
30  int max_iter = 10;

34  private:

39  fp_t step,
40  fp_t f_0, fp_t g_0,

42  {
43  using std::abs;
44  return (f_t <= f_0 + step * wolfe_c1 * g_0) &&
45  (abs(g_t) <= wolfe_c2 * abs(g_0));

48  public:
50  void line_search(Args &arg) override
51  {
52  using std::abs;
53  using std::max;
54  using std::min;
55  using namespace internal::MTLineSearch;
56  optim_assert(arg.step > 0, "step must be positive.");
57  iter = 0; // reset iter
58  fp_t g_0, g_t; // = grad_0.T * d;
59  // check wolfe condition
60  arg.step_forward(this->prob.get());
61  arg.update_cur_loss(this->prob.get());
62  arg.update_cur_grad(this->prob.get());
63  // update subgradient if needed
64  if constexpr (use_prox)
65  {
66  arg.update_prev_grad_map(this->prob.get());
67  arg.update_cur_grad_map(this->prob.get());
68  arg.tmp = arg.cur_x - arg.prev_x;
69  g_0 = BMO_MAT_DOT_PROD(arg.prev_grad_map, arg.tmp) /
70  arg.step;
71  g_t = BMO_MAT_DOT_PROD(arg.cur_grad_map, arg.tmp) /
72  arg.step;
73  }
74  else
75  {
76  g_0 = BMO_MAT_DOT_PROD(arg.prev_grad, arg.direction);
77  g_t = BMO_MAT_DOT_PROD(arg.cur_grad, arg.direction);
78  }
79  optim_assert(g_0 < 0, "g_0 must be negtive.");
80  if (check_wolfe_condition(
81  arg.step, arg.prev_loss, g_0, arg.cur_loss, g_t))
82  return;
83  bool use_auxiliary = 1, brackt = 0;
84  fp_t step_max = arg.step * 10,
85  step_min = this->min_step;
86  fp_t width = this->max_step - this->min_step,
87  width_old = 2. * width;
88  FuncVal<fp_t, use_prox>
89  a_l(0, arg.prev_loss, g_0),
90  a_t(arg.step, arg.cur_loss, g_t),
91  a_u(0, arg.prev_loss, g_0);
92  const int n = BMO_ROWS(arg.cur_x),
93  m = BMO_COLS(arg.cur_x);
94  Mat<fp_t> best_x(n, m), best_grad(n, m), best_gmap;
95  if constexpr (use_prox)
96  BMO_RESIZE(best_gmap, n, m);
97  g_t = wolfe_c1 * g_0; // gtest in origin code
98  logger.info("[MTLS] start line-search.");
99  for (iter = 1; iter <= max_iter; iter++)
100  {
101  if (use_auxiliary)
102  {
103  a_l.phi_to_psi(g_t);
104  a_t.phi_to_psi(g_t);
105  a_u.phi_to_psi(g_t);
106  arg.step = select_trial_value<fp_t>(
107  brackt, a_l, a_u, a_t, step_min, step_max);
108  // Reset the function and derivative values for f.
109  a_l.psi_to_phi(g_t);
110  a_t.psi_to_phi(g_t);
111  a_u.psi_to_phi(g_t);
112  }
113  else
114  arg.step = select_trial_value(
115  brackt, a_l, a_u, a_t, step_min, step_max);
116  // Force the step to be within the bounds
117  arg.step = std::min(std::max(arg.step, this->min_step), this->max_step);
118  // ----------------- Updating Method -----------------
119  if (a_t.val > a_l.val)
120  // case U1 & a
121  a_u.swap(a_t); // shrink interval
122  else
123  {
124  if (std::signbit(a_t.deriv) !=
125  std::signbit(a_l.deriv))
126  // a_t and a_l should be endpoints
127  a_u.swap(a_l);
128  a_l.swap(a_t); // case U2 & b
129  BMO_SWAP(best_x, arg.cur_x);
130  BMO_SWAP(best_grad, arg.cur_grad);
131  if constexpr (use_prox)
132  best_gmap = std::move(arg.cur_grad_map);
133  }
134  if (check_wolfe_condition(
135  a_l.arg, arg.prev_loss, g_0, a_l.val, a_l.deriv))
136  { // that's what we want!
137  status = 0;
138  goto over;
139  }
140  // update the 'interval of uncertainty'
141  if (brackt)
142  {
143  if (abs(a_u.arg - a_l.arg) > 0.66 * width_old)
144  // Decide if a bisection step is needed.
145  arg.step = (a_l.arg + a_u.arg) / 2;
146  width_old = width;
147  width = abs(a_u.arg - a_l.arg);
148  if (step_max < Constant::sqrt_eps)
149  OPTIM_UNLIKELY
150  { // minimal is too close to origin point!
151  optim_assert(a_l.val <= arg.prev_loss,
152  "f_t should be less than f_0. Please check your loss and grad func.");
153  status = -1;
154  goto over;
155  }
156  step_max = max(a_l.arg, a_u.arg);
157  step_min = min(a_l.arg, a_u.arg);
158  if ((arg.step > step_max || arg.step < step_min) ||
159  step_max - step_min <= 1e-4 * step_max) [[unlikely]]
160  arg.step = (step_max + step_min) / 2;
161  }
162  else
163  { // step too small, expand interval.
164  step_min = arg.step + 1.1 * (arg.step - a_l.arg);
165  step_max = arg.step + 5.0 * (arg.step - a_l.arg);
166  }
167  // log
168  if (iter % 10 == 0)
169  {
170  logger.info("[MTLS] iter: {:<3d} brackt:{:<5} use_auxilary: {:<5}\nstep_min: {:<10g} | a_l: {:<10g} | a_t: {:<10g} | a_u: {:<10g} | step_max: {:<10g}", iter, brackt, use_auxiliary, step_min, a_l.arg, a_t.arg, a_u.arg, step_max);
171  }
172  else
173  {
174  // f_t <= f_0 + step * wolfe_c1 * g_0
175  logger.trace("[MTLS] iter: {:<3d} brackt:{:<5} use_auxilary: {:<5}\n| a_l: {:<10g} | a_t: {:<10g} | a_u: {:<10g} |\n| f_l: {:<10g} | f_t: {:<10g} | f_u: {:<10g} |\n| g_l: {:<10g} | g_t: {:<10g} | g_u: {:<10g} |", iter, brackt, use_auxiliary, a_l.arg, a_t.arg, a_u.arg, a_l.val, a_t.val, a_u.val, a_l.deriv, a_t.deriv, a_u.deriv);
176  }
177  // compute new step
178  a_t.update_val(this->prob.get(), arg);
179  // check if switch to "Modified Updating Method"
180  if (use_auxiliary &&
181  a_t.val <= arg.prev_loss + g_t && // psi(a_t) <= 0
182  a_t.deriv >= 0) // phi'(a_t) >= 0
183  use_auxiliary = 0;
184  }
185  status = 1;
186  logger.warn("[MTLS] reaches max_iter: {}.", max_iter);
187  optim_assert(a_l.val <= arg.prev_loss,
188  "cur_loss should be less than prev_loss. Please check your loss, grad func.");
189  over:
190  arg.step = a_l.arg;
191  arg.cur_loss = a_l.val;
192  arg.cur_x = std::move(best_x);
193  arg.cur_grad = std::move(best_grad);
194  if constexpr (use_prox)
195  arg.prev_grad_map = std::move(best_gmap);
196  }
197 
198  bool success() const override { return status == 0; }
199 
200  int n_iter() const { return iter; };
201  };
